# 输出题精选

> 高频 JavaScript 输出题，考察变量提升、this 指向、Promise 执行顺序、作用域等核心知识点。

---

## 题 1: 变量提升

```js
console.log(a); // undefined
var a = 1;

console.log(b); // ReferenceError
let b = 2;
```

---

## 题 2: this 指向

```js
const obj = {
  name: '张三',
  say: function() { console.log(this.name); },
  say2: () => { console.log(this.name); }
};
obj.say();  // '张三'
obj.say2(); // undefined（箭头函数的 this 指向定义时外层，即 window）
```

---

## 题 3: Promise 执行顺序

```js
console.log('start');
setTimeout(() => console.log('timeout'), 0);
Promise.resolve().then(() => console.log('promise'));
console.log('end');
// start → end → promise → timeout
```

---

## 题 4: 作用域

```js
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}
// 3 3 3（var 没有块作用域）

for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 0);
}
// 0 1 2（let 有块作用域）
```

---

## 题 5: 连续赋值

```js
var a = { n: 1 };
var b = a;
a.x = a = { n: 2 };
console.log(a.x); // undefined
console.log(b.x); // { n: 2 }
```

---

## 题 6: 对象键

```js
const a = {};
const b = { key: 'b' };
const c = { key: 'c' };

a[b] = 123; // a['[object Object]'] = 123
a[c] = 456; // a['[object Object]'] = 456 —— 覆盖

console.log(a[b]); // 456
```

---

## 题 7: async/await 执行顺序

```js
async function async1() {
  console.log('async1 start');
  await async2();
  console.log('async1 end');
}
async function async2() {
  console.log('async2');
}
console.log('script start');
setTimeout(() => console.log('setTimeout'), 0);
async1();
new Promise((resolve) => {
  console.log('promise1');
  resolve();
}).then(() => console.log('promise2'));
console.log('script end');

// script start → async1 start → async2 → promise1 → script end
// → async1 end → promise2 → setTimeout
```

## 题 8: Promise 输出题合集

### 8.1

```js
const promise = new Promise((resolve, reject) => {
  console.log(1);
  resolve();
  console.log(2);
});

promise.then(() => {
  console.log(3);
});

console.log(4);
// 输出: 1 2 4 3
```

### 8.2

```js
const promise = new Promise((resolve, reject) => {
  console.log(1);
  setTimeout(() => {
    console.log(2);
    resolve();
    console.log(3);
  });
});

promise.then(() => {
  console.log(4);
});

console.log(5);
// 输出: 1 5 2 3 4
```

### 8.3

```js
const promise1 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve();
  }, 1000);
});
const promise2 = promise1.catch(() => {
  return 2;
});

console.log('promise1', promise1);
console.log('promise2', promise2);

setTimeout(() => {
  console.log('promise1', promise1);
  console.log('promise2', promise2);
}, 2000);
// 同步输出: promise1 Promise{<pending>}, promise2 Promise{<pending>}
// 2秒后输出: promise1 Promise{<fulfilled>}, promise2 Promise{<fulfilled>}
```

### 8.4

```js
async function m() {
  const n = await 1;
  console.log(n);
}

m();
console.log(2);
// 输出: 2 1
```

### 8.5

```js
async function m() {
  const n = await 1;
  console.log(n);
}

(async () => {
  await m();
  console.log(2);
})();

console.log(3);
// 输出: 3 1 2
```

### 8.6

```js
async function m1() {
  return 1;
}

async function m2() {
  const n = await m1();
  console.log(n);
  return 2;
}

async function m3() {
  const n = m2();
  console.log(n);
  return 3;
}

m3().then((n) => {
  console.log(n);
});

m3();

console.log(4);
// 输出: Promise{<pending>} Promise{<pending>} 4 1 3 1 3
```

### 8.7

```js
Promise.resolve(1).then(2).then(Promise.resolve(3)).then(console.log);
// 输出: 1（then 中传入非函数会发生穿透）
```

### 8.8

```js
var a;
var b = new Promise((resolve, reject) => {
  console.log('promise1');
  setTimeout(() => {
    resolve();
  }, 1000);
})
  .then(() => {
    console.log('promise2');
  })
  .then(() => {
    console.log('promise3');
  })
  .then(() => {
    console.log('promise4');
  });

a = new Promise(async (resolve, reject) => {
  console.log(a);
  await b;
  console.log(a);
  console.log('after1');
  await a;
  resolve(true);
  console.log('after2');
});

console.log('end');
// 同步输出: promise1, undefined, end
// 1秒后: promise2, promise3, promise4, Promise{<pending>}, after1
// after2 永远不会输出（await a 死锁）
```

### 8.9

```js
async function async1() {
  console.log('async1 start');
  await async2();
  console.log('async1 end');
}
async function async2() {
  console.log('async2');
}

console.log('script start');

setTimeout(function () {
  console.log('setTimeout');
}, 0);

async1();

new Promise(function (resolve) {
  console.log('promise1');
  resolve();
}).then(function () {
  console.log('promise2');
});
console.log('script end');

// script start → async1 start → async2 → promise1 → script end
// → async1 end → promise2 → setTimeout
```
